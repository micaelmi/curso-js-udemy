Strings

        let text = "Um texto"; 

        console.log(text[4]); // e

        console.log(text.charAt(0)); // U

        console.log(text.concat(' legal')); // Um texto legal

        console.log(text.indexOf('texto')); // 3

        console.log(text.indexOf('texto', 4)); // -1 (não encontrou depois da
        posição 4)

        console.log(text.lastIndexOf('t')); // 6

        console.log(text.match(/[a-z]/g)); // ['m', 't', 'e', 'x', 't', 'o']

        console.log(text.search(/x/)); // 5 (índice) da letra x

        console.log(text.replace('Um', 'Outro')); // Outro texto

        console.log(text.length); // 7

        console.log(text.slice(3, 5)); // tex (recorta um pedaço)

        console.log(text.slice(-5)); // texto (recorta um pedaço do final pro
        começo)

        console.log(text.split('')); // ['U', 'm', ' ', 't', 'e', 'x', 't',
        'o']

        console.log(text.toUpperCase()); // UM TEXTO

        console.log(text.toLowerCase()); // um texto

Arrays

        students[1] = 'Alex' // altera posição 1 

        students[students.length] = 'Clair' // adiciona no final

        function createPerson(name, lastname, age) { 

          return { 

            name, 

            lastname, 

            age, 

            speak() { 

              console.log(`Hello, I'm ${this.name}! My age is ${this.age}`); 

            }, 

            ageIncrement() { 

              this.age++; 

            } 

          } 

        } 

        const person2 = createPerson('Micael', 'Miranda', 18); 

        person2.speak(); 

        person2.ageIncrement(); 

        person2.speak();
