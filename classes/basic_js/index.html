Continue e Break

        // Continue -> pula para a próxima repetição do laço 

        // Break -> sai do laço de repetição 

        const numeros = [1,2,3,4,5] 

        for (let num of numeros) { 

          if(num === 2) { 

            continue; 

          } 

          if(num === 4) { 

            break; 

          } 

          console.log(num); 

        } 

        

        // saída -> 1, 3

        > EXEMPLO: 

        function soma(x, y) { 

          if(typeof x !== 'number' || typeof y !== 'number') { 

            throw('x e y devem ser números');

            // throw new Error('x e y devem ser números');

          } 

          return x + y; 

        } 

        try { 

          console.log(soma(1, 2)); // resultado: 3 

          console.log(soma('1', 2)); // cria um erro 

        } catch(error) { 

          console.log(error); // mensagem de erro da função 

        }

setInterval e setTimeout

        function mostraHora() { 

          let data = new Date(); 

          return data.toLocaleTimeString('pt-BR') 

        }

        const timer = setInterval(function() { 

          console.log(mostraHora()); 

        },1000); // executa a função a cada 1 segundo 

        

        setTimeout(function() { 

          clearInterval(timer); 

          console.log('Timer parado!'); 

        }, 5000); // depois de 5s, para a função de timer
