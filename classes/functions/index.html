        function newFunction() { 

          let total = 0; 

          for (let argument of arguments) { 

            total += argument; 

          } 

          console.log(total); 

        } 

        newFunction('Valor', 'Micael', 1, 2, 3);

Retornos da função

        function soma(a, b) { 

          return a+b; 

        } 

        console.log(soma(1, 2)); 

        

        function newPerson(name, lastname) { 

          return { name, lastname}; 

        } 

        const p1 = newPerson('Micael', 'Inacio'); 

        console.log(p1); 

        

        function showPhrase(start) { 

          function showRest(end) { 

            return start + ' ' + end; 

          } 

          return showRest; 

        } 

        const speak = showPhrase('Hello'); 

        const rest = speak(' world'); 

        console.log(rest);
